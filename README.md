### This program adds two numbers given by the user

# Taking input from the user
num1 = input("Enter first number: ")
num2 = input("Enter second number: ")

# Convert input from string to integer
num1 = int(num1)
num2 = int(num2)

# Add the two numbers
sum = num1 + num2

# Display the result
print("The sum is:", sum)
